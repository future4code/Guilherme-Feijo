LABOOK

ESTRUTURA DE DADOS

  usuarios
    id
    nome
    email
    senha

  posts
    idUsuario (FK)
    id
    foto
    descricao
    dataCriacao
    tipo: "normal" | "evento"
    
  curtidas
    idUsuario (FK)
    idPost (FK)

  comentarios
    id
    idUsuario (FK)
    idPost (FK)
    texto

FUNCIONALIDADES

  cadastrar: (dadosUsuario) => token

  logar: userCredentials => token

  criarPost: (dadosPost, token) => mensagem

  buscarPostPorId: id => post


BANCO DE DADOS

  CREATE TABLE labook_users (
    id VARCHAR(255) PRIMARY KEY,
      nome VARCHAR(255) NOT NULL,
      email VARCHAR(255) UNIQUE NOT NULL,
      senha VARCHAR(255) NOT NULL
  );

  CREATE TABLE labook_posts (
    id VARCHAR(255) PRIMARY KEY,
      foto VARCHAR(255) NOT NULL,
      descricao VARCHAR(255) NOT NULL,
      dataCriacao DATETIME NOT NULL,
      tipo ENUM ("NORMAL" , "EVENT") DEFAULT "NORMAL",
      userId VARCHAR(255),
      FOREIGN KEY (userId) REFERENCES labook_users(id)
  );
